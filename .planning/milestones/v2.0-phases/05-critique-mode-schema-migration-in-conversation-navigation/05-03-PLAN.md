    const STORAGE_KEY = 'critique-run-timestamps';
    const WINDOW_MS = 60 * 60 * 1000;  // 1 hour
    const LIMIT = 5;

    export function getRecentRunCount() {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        const arr = raw ? JSON.parse(raw) : [];
        const cutoff = Date.now() - WINDOW_MS;
        const filtered = arr.filter((ts) => ts > cutoff);
        if (filtered.length !== arr.length) {
          localStorage.setItem(STORAGE_KEY, JSON.stringify(filtered));
        }
        return filtered.length;
      } catch {
        return 0;
      }
    }

    export function recordCritiqueRun() {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        const arr = raw ? JSON.parse(raw) : [];
        const cutoff = Date.now() - WINDOW_MS;
        const filtered = arr.filter((ts) => ts > cutoff);
        filtered.push(Date.now());
        localStorage.setItem(STORAGE_KEY, JSON.stringify(filtered));
      } catch {
        /* best-effort */
      }
    }

    export function shouldReconfirm() {
      return getRecentRunCount() >= LIMIT;
    }

    async sendCritiqueStream(conversationId, critiqueInstruction, slots, onEvent) {
      const form = new FormData();
      form.append('critique_instruction', critiqueInstruction);
      slots.forEach((slot, i) => {
        if (slot && slot.file) {
          form.append(`file_slot_${i}`, slot.file, slot.file.name);
        }
      });
      const response = await fetch(
        `${API_BASE}/api/conversations/${conversationId}/critique/stream`,
        { method: 'POST', body: form }   // NO Content-Type header — browser sets multipart boundary
      );
      if (!response.ok) {
        const detail = await response.json().catch(() => ({ detail: 'Failed to send critique' }));
        throw new Error(detail.detail || 'Failed to send critique');
      }
      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        const chunk = decoder.decode(value);
        const lines = chunk.split('\n');
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            try {
              const event = JSON.parse(line.slice(6));
              onEvent(event.type, event);
            } catch (e) {
              console.error('Failed to parse SSE event:', e);
            }
          }
        }
      }
    },

    import { useState, useMemo } from 'react';
    import DropZoneSlot from './DropZoneSlot';
    import { MAX_CRITIQUE_FILE_BYTES, readFileAsText } from '../utils/download';
    import { shouldReconfirm, recordCritiqueRun } from '../utils/critiqueRateLimit';
    import './CritiqueWelcome.css';

    // MUST mirror PROFILES["quality"]["council_models"] order in backend/config.py
    const QUALITY_SLOT_MODELS = [
      'openai/gpt-5.5',
      'anthropic/claude-opus-4.7',
      'google/gemini-3.1-pro-preview',
    ];

    const HEURISTIC_TOKENS_PER_CHAR = 0.25;
    const TOKEN_CAP = 150_000;

    function formatKB(bytes) { return (bytes / 1024).toFixed(1); }

    function estimateCostRange(slots, instruction) {
      const activeChars = slots.reduce((acc, s) => acc + (s ? s.content.length : 0), 0) + instruction.length;
      const n = slots.filter(Boolean).length;
      if (n === 0) return null;
      const inputTokens = (activeChars * HEURISTIC_TOKENS_PER_CHAR) * n;
      const outputTokens = 2000 * n + 1500;  // Stage 1 + Stage 2 + Stage 3 rough output total per RESEARCH §4.2
      // Worst-case anchor: Opus 4.7 rates — $5/M input, $25/M output (RESEARCH §4.2)
      const baseLow = (inputTokens * 5 / 1_000_000) + (outputTokens * 25 / 1_000_000);
      const low = baseLow * 0.7;
      const high = baseLow * 1.3;
      return { low: low.toFixed(2), high: high.toFixed(2), tokens: Math.round(inputTokens) };
    }

    export default function CritiqueWelcome({ onSubmit, isLoading }) {
      const [slots, setSlots] = useState([null, null, null]);
      const [instruction, setInstruction] = useState('');
      const [slotErrors, setSlotErrors] = useState([null, null, null]);
      const [tokenCapError, setTokenCapError] = useState(null);
      const [showReconfirm, setShowReconfirm] = useState(false);

      const cost = useMemo(() => estimateCostRange(slots, instruction), [slots, instruction]);
      const hasFile = slots.some(Boolean);
      const hasInstruction = instruction.trim().length > 0;
      const hasError = slotErrors.some(Boolean) || tokenCapError;
      const canSubmit = hasFile && hasInstruction && !hasError && !isLoading;

      const handleFile = async (idx, file) => {
        const nextErrors = [...slotErrors];
        nextErrors[idx] = null;
        // Size validation
        if (file.size > MAX_CRITIQUE_FILE_BYTES) {
          nextErrors[idx] = `${file.name} is ${formatKB(file.size)} KB — maximum is 750 KB. Trim the file or split it.`;
          setSlotErrors(nextErrors);
          return;
        }
        // Extension validation
        const lower = file.name.toLowerCase();
        if (!(lower.endsWith('.md') || lower.endsWith('.txt'))) {
          nextErrors[idx] = `${file.name} must be .md or .txt. PDF and DOCX are not supported.`;
          setSlotErrors(nextErrors);
          return;
        }
        let content;
        try {
          content = await readFileAsText(file);
        } catch (e) {
          nextErrors[idx] = `Could not read ${file.name}: ${e.message}`;
          setSlotErrors(nextErrors);
          return;
        }
        // Normalize line endings client-side too (server also normalizes)
        content = content.replace(/\r\n/g, '\n').replace(/\r/g, '\n');
        const nextSlots = [...slots];
        nextSlots[idx] = { file, modelId: QUALITY_SLOT_MODELS[idx], name: file.name, size: file.size, content };
        setSlots(nextSlots);
        setSlotErrors(nextErrors);
        setTokenCapError(null);
      };

      const handleRemove = (idx) => {
        const nextSlots = [...slots];
        nextSlots[idx] = null;
        setSlots(nextSlots);
        const nextErrors = [...slotErrors];
        nextErrors[idx] = null;
        setSlotErrors(nextErrors);
        setTokenCapError(null);
      };

      const dispatch = () => {
        recordCritiqueRun();
        onSubmit(instruction.trim(), slots);
      };

      const handleSubmit = (e) => {
        e.preventDefault();
        if (!canSubmit) return;
        if (cost && cost.tokens > TOKEN_CAP) {
          setTokenCapError(`These files total ≈${Math.round(cost.tokens / 1000)}K tokens, above the 150K cap. Remove one file or trim the largest.`);
          return;
        }
        if (shouldReconfirm()) {
          setShowReconfirm(true);
          return;
        }
        dispatch();
      };

      return (
        <div className="critique-welcome">
          <h1 className="critique-welcome__title">Critique research</h1>
          <p className="critique-welcome__lead">
            Upload up to three deep-research files. The council will read them all, critique each one, then synthesise.
          </p>

          <form onSubmit={handleSubmit} className="critique-welcome__form">
            {QUALITY_SLOT_MODELS.map((modelId, i) => (
              <DropZoneSlot
                key={modelId}
                modelId={modelId}
                slot={slots[i]}
                onFile={(file) => handleFile(i, file)}
                onRemove={() => handleRemove(i)}
                error={slotErrors[i]}
              />
            ))}

            <textarea
              className="critique-welcome__instruction"
              placeholder="Identify factual errors, missing perspectives, and weak arguments in these research files…"
              value={instruction}
              onChange={(e) => setInstruction(e.target.value)}
              required
              rows={4}
            />

            {tokenCapError && (
              <div className="critique-welcome__error">{tokenCapError}</div>
            )}

            <div className="critique-welcome__cost-row">
              {cost && (
                <span className="critique-welcome__cost">
                  Estimated upstream: ${cost.low}–${cost.high} (billed to your provider keys, not OpenRouter)
                </span>
              )}
              <button
                type="submit"
                className="critique-welcome__submit"
                disabled={!canSubmit}
                title={!canSubmit ? 'Attach at least one file and write a critique instruction to submit' : undefined}
              >
                Submit critique
              </button>
            </div>
          </form>

          {showReconfirm && (
            <div className="critique-welcome__reconfirm" role="dialog" aria-modal="true">
              <p>You have launched 5 critiques in the last hour. Continue?</p>
              <button type="button" onClick={() => { setShowReconfirm(false); dispatch(); }}>Continue</button>
              <button type="button" onClick={() => setShowReconfirm(false)}>Cancel</button>
            </div>
          )}
        </div>
      );
    }
